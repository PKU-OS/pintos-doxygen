14/** Opens a file for the given INODE, of which it takes ownership,

17struct file *
18file_open (struct inode *inode)
19{
20 struct file *file = calloc (1, sizeof *file);
21 if (inode != NULL && file != NULL)
22 {
23 file->inode = inode;
24 file->pos = 0;
25 file->deny_write = false;
